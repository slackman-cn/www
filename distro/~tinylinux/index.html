Shell & ISO

man 2 // read,write,execve,fork,wait

#include <unistd.h>
#include <sys/wait.h>

int main()
{
    char cmd[255];
    for (;;) {
        write(1, "# ", 2);
        int count = read(0, cmd, 255);
        cmd[count-1] = 0;
        pid_t fork_result = fork();
        if (fork_result == 0) {
            execve(cmd, 0, 0);
            break;
        } else {
            siginfo_t info;
            waitid(P_ALL, 0, &info, WEXITED);
        }
    }
}
