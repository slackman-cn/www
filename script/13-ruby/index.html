# 实例变量@xx， 类变量@@yy所有实例共享
class HelloWorld
  @@count = 0
  def initialize(myname="Ruby")
    @name=myname
    @@count+=1
  end

  def hello
    puts "Hello #{@name}  / #{@@count}"
  end
end

hx = HelloWorld.new("Tomas")
hx2 = HelloWorld.new
hx.hello
