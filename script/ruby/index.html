HelloWorld

class Person
  # 静态变量
  @@cnt = 1

  # 属性，实例变量
  attr_accessor :name, :age

  # 初始化方法
  def initialize(name, age)
    @name = name
    @age = age
  end

  # 定义方法
  def greet
    "My name is #{@name} and I am #{@age} years old."
  end
end

person = Person.new("Alice", 30)
puts person.greet

# 实例变量@xx， 类变量@@yy所有实例共享
class HelloWorld
  @@count = 0
  def initialize(myname="Ruby")
    @name=myname
    @@count+=1
  end

  def hello
    puts "Hello #{@name}  / #{@@count}"
  end
end

hx = HelloWorld.new("Tomas")
hx2 = HelloWorld.new
hx.hello
